Here's the implementation of the Cashier class in Python:

```python
from typing import List

class Cashier:

    def __init__(self, n: int, discount: int, products: List[int], prices: List[int]):
        self.n = n
        self.discount = discount
        self.products = products
        self.prices = prices
        self.customer_count = 0

    def getBill(self, product: List[int], amount: List[int]) -> float:
        self.customer_count += 1
        subtotal = 0
        
        for i in range(len(product)):
            product_id = product[i]
            product_price = self.prices[self.products.index(product_id)]
            subtotal += product_price * amount[i]
        
        if self.customer_count % self.n == 0:
            discount_amount = (subtotal * self.discount) / 100
            total = subtotal - discount_amount
        else:
            total = subtotal
        
        return total
```

You can create an instance of the Cashier class and call the `getBill` method to calculate the final total of the bill with the discount applied (if any), as shown in the example:

```python
cashier = Cashier(3, 50, [1, 2, 3, 4, 5, 6, 7], [100, 200, 300, 400, 300, 200, 100])

# Example usage
print(cashier.getBill([1, 2], [1, 2]))  # 500.0
print(cashier.getBill([3, 7], [10, 10]))  # 4000.0
print(cashier.getBill([1, 2, 3, 4, 5, 6, 7], [1, 1, 1, 1, 1, 1, 1]))  # 800.0
print(cashier.getBill([4], [10]))  # 4000.0
print(cashier.getBill([7, 3], [10, 10]))  # 4000.0
print(cashier.getBill([7, 5, 3, 1, 6, 4, 2], [10, 10, 10, 9, 9, 9, 7]))  # 7350.0
print(cashier.getBill([2, 3, 5], [5, 3, 2]))  # 2500.0
```
