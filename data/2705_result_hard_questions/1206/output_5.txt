class Node:

        self.val = val
        self.right = None

class Skiplist:

        self.head = Node(float('-inf'))
        self.tail = Node(float('inf'))
        self.head.right = self.tail
        self.levels = [self.head]  # List to keep track of the levels

    def search(self, target: int) -> bool:
        curr = self.head
        while curr:
            if curr.val == target:
                return True
            elif curr.right.val > target:
                curr = curr.down
            else:
                curr = curr.right
        return False
