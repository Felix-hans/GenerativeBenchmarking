Here is the code implementation for the given task:

```go
import "fmt"

func zeroFilledSubarray(nums []int) int64 {
    count := int64(0)
    length := len(nums)
    for i := 0; i < length; i++ {
        if nums[i] == 0 {
            count++
        }
        for j := i + 1; j < length; j++ {
            if nums[j] == 0 {
                count++
            } else {
                break
            }
        }
    }
    return count
}
```

Example usage:
```go
func main() {
    nums := []int{1, 3, 0, 0, 2, 0, 0, 4}
    result := zeroFilledSubarray(nums)
    fmt.Println(result) // Output: 6
}
```
