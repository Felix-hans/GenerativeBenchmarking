from typing import List

class Solution:
    def isPossibleToCutPath(self, grid: List[List[int]]) -> bool:
        def dfs(row, col, grid):
            if row == len(grid) - 1 and col == len(grid[0]) - 1:
                return True

            grid[row][col] = -1 # mark the cell as visited

            # explore the neighbors: (row + 1, col) and (row, col + 1)
            directions = [(1, 0), (0, 1)]
            for dx, dy in directions:
                new_row = row + dx
                new_col = col + dy

                if new_row < len(grid) and new_col < len(grid[0]) and grid[new_row][new_col] in [1, -1]:
                    if dfs(new_row, new_col, grid):
                        return True

            return False
        
        # if there is already no path from (0, 0) to (m-1, n-1), return True
        if not dfs(0, 0, grid):
            return True

        # try to flip one cell and check if it makes the matrix disconnected
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    grid[i][j] = 0

                    if not dfs(0, 0, grid):
                        return True

                    grid[i][j] = 1

        return False
